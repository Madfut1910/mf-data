<button
  class="download-face-btn"
  onclick="event.preventDefault(); event.stopPropagation(); downloadFaceImage(
    this.closest('.card').querySelector('.player-face'),
    '${name.replace(/'/g, '').replace(/\s+/g, '_')}_FACE.png'
  )"
>
  <svg viewBox="0 0 24 24">
    <path d="M12 12c2.7 0 5-2.3 5-5s-2.3-5-5-5-5 2.3-5 5 2.3 5 5 5zm0 2c-3.3 0-10 1.7-10 5v3h20v-3c0-3.3-6.7-5-10-5z"/>
  </svg>
</button>

  <script src="https://cdnjs.cloudflare.com/ajax/libs/html2canvas/1.4.1/html2canvas.min.js"></script>

  <script>
    function downloadCardImage(cardElement, fileName) {
      const downloadBtn = cardElement.querySelector('.download-btn');
      const faceBtn = cardElement.querySelector('.download-face-btn');

      // Add animation
      if (downloadBtn) {
        downloadBtn.classList.add('downloading');
        setTimeout(() => downloadBtn.classList.remove('downloading'), 600);
      }

      if (downloadBtn) downloadBtn.style.display = 'none';
      if (faceBtn) faceBtn.style.display = 'none';

      const totwElement = cardElement.querySelector('.totwNumber');
      const prevTotwDisplay = totwElement ? totwElement.style.display : null;
      if (totwElement) totwElement.style.display = 'none';

      const chemElement = cardElement.querySelector('.specialChem');
      const prevChemDisplay = chemElement ? chemElement.style.display : null;
      if (chemElement) chemElement.style.display = 'none';

      const rect = cardElement.getBoundingClientRect();

      html2canvas(cardElement, {
        backgroundColor: null,
        useCORS: true,
        scale: 3,
        width: rect.width,
        height: rect.height
      }).then(canvas => {
        if (downloadBtn) downloadBtn.style.display = '';
        if (faceBtn) faceBtn.style.display = '';
        if (totwElement) totwElement.style.display = prevTotwDisplay;
        if (chemElement) chemElement.style.display = prevChemDisplay;

        canvas.toBlob(blob => {
          const a = document.createElement('a');
          a.href = URL.createObjectURL(blob);
          a.download = fileName;
          a.click();
        }, 'image/png');
      });
    }

    function downloadFaceImage(imgElement, fileName) {
      // Add animation to face button
      const faceBtn = imgElement.closest('.card').querySelector('.download-face-btn');
      if (faceBtn) {
        faceBtn.classList.add('downloading');
        setTimeout(() => faceBtn.classList.remove('downloading'), 600);
      }

      const canvas = document.createElement('canvas');
      const ctx = canvas.getContext('2d');
      canvas.width = imgElement.naturalWidth || imgElement.width;
      canvas.height = imgElement.naturalHeight || imgElement.height;
      ctx.drawImage(imgElement, 0, 0);
      canvas.toBlob(blob => {
        const a = document.createElement('a');
        a.href = URL.createObjectURL(blob);
        a.download = fileName;
        a.click();
      }, 'image/png');
    }
  </script>
